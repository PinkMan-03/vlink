cmake_minimum_required(VERSION 3.15...3.31)

project(example_plugin_schema LANGUAGES CXX)

include(GNUInstallDirs)

if(NOT TARGET vlink::vlink)
  find_package(vlink QUIET)
endif()
if(NOT TARGET vlink::vlink)
  message(FATAL_ERROR "Can not find_package for vlink.")
endif()

# This example is conceptual -- it demonstrates the SchemaPluginInterface API without building a .so (the real plugin
# requires protobuf).
add_executable(${PROJECT_NAME} plugin_schema.cc)
target_link_libraries(${PROJECT_NAME} vlink::all)

install(TARGETS ${PROJECT_NAME} RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR})
